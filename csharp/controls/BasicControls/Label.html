Introduction to UI Controls

Example: Adding a Label and a Button

1. Modify Form.Designer.cs


  private System.Windows.Forms.Label label1;
  private System.Windows.Forms.Button button1;
  
  private void InitializeComponent()
  {
      this.label1 = new System.Windows.Forms.Label();
      this.button1 = new System.Windows.Forms.Button();
      
      // Label properties
      this.label1.Text = "Hello, VSCode!";
      this.label1.Location = new System.Drawing.Point(20, 20);
      
      // Button properties
      this.button1.Text = "Click Me";
      this.button1.Location = new System.Drawing.Point(20, 60);
      this.button1.Click += new System.EventHandler(this.Button1_Click);
  
      // Add controls to the form
      this.Controls.Add(this.label1);
      this.Controls.Add(this.button1);
  }

2. Modify Form.cs (Event Handler)


  private void Button1_Click(object sender, EventArgs e)
  {
      this.label1.Text = "Button Clicked!";
  }

  private System.Windows.Forms.NumericUpDown numericUpDown1;
