Form Files

This page explains how to add a form file to your project. Unlike Visual Studio, you need to manually create both SampleForm.cs and SampleForm.Designer.cs with the appropriate templates. In this example, we'll assume the project name is SampleProject, and the form file is named SampleForm.cs.

1. Add two new files to your project and name them SampleForm.cs and SampleForm.Designer.cs.

2. Manually add the following template code:

The recommended templates for class files are as follows:

SampleForm.cs 
using System;
using System.Collections.Generic;
using System.ComponentModel;
using System.Data;
using System.Drawing;
using System.Linq;
using System.Text;
using System.Threading.Tasks;
using System.Windows.Forms;

namespace SampleProject
{
    public partial class SampleForm : Form
    {
        public SampleForm()
        {
            InitializeComponent(); // Ensure this matches the method in SampleForm.Designer.cs
        }
    }
}

SampleForm.Designer.cs 
namespace SampleProject
{
    partial class SampleForm
    {
        /// 
        /// Holds the form's components. Dispose them properly if modified.
        /// 
        private System.ComponentModel.IContainer components = null;

        /// 
        /// Clean up any resources being used.
        /// 
        /// 
        /// True to release managed resources; otherwise, false.
        /// 
        protected override void Dispose(bool disposing)
        {
            if (disposing && (components != null))
            {
                components.Dispose();
            }
            base.Dispose(disposing);
        }

        #region Template code for Windows Form Designer support (manually added)

        /// 
        /// Initializes the form's components. 
        /// Modify this method if you add controls.
        /// 
        private void InitializeComponent()
        {
            this.components = new System.ComponentModel.Container();
            this.AutoScaleMode = System.Windows.Forms.AutoScaleMode.Font;
            this.ClientSize = new System.Drawing.Size(800, 450);
            this.Text = "SampleForm";
        }

        #endregion
    }
}
